public class AccountTriggerHandler {

    
    public static void CreateAccounts(List<Account> accountsTriggered){
        /* The method must accept a List of Account objects
		 For each Account record, before saving, 
ensure that the ShippingState field has the same value as the BillingState field
*/    

        List<Account> StateAcc=new List<Account>();
      
         // System.debug(' >> Handler   : '+accountsTriggered);
        
        for (Account a:accountsTriggered) {
            a.ShippingState=a.BillingState;
          
         
        }
        
       System.debug('Size of Account Updated : '+ accountsTriggered.size());
      
        
}
}

Trigger

trigger AccountTrigger on Account (before insert) {

    
    if (Trigger.isBefore && Trigger.isInsert){
    
        
          AccountTriggerHandler.CreateAccounts(Trigger.New);
    
    }
  
    
}

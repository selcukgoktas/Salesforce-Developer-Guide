NULL and Operator ! = not Equal


// NULLS FIRST

SELECT Name,id,Industry,AnnualRevenue FROM Account ORDER BY AnnualRevenue DESC NULLS FIRST
// NULLS LAST
SELECT Name,id,Industry,AnnualRevenue FROM Account ORDER BY AnnualRevenue DESC NULLS LAST


// WHERE != null
SELECT Name,id,Industry,AnnualRevenue FROM Account WHERE AnnualRevenue != null ORDER BY AnnualRevenue DESC

SELECT Name,id,Industry,AnnualRevenue FROM Account ORDER BY AnnualRevenue
// AnnualRevenue ordered

SELECT Name,id,Industry,AnnualRevenue FROM Account ORDER BY AnnualRevenue DESC
// AnnualRevenue azalarak

SELECT Name,id,Industry,AnnualRevenue FROM Account ORDER BY AnnualRevenue ASC
// AnnualRevenue artarak

SELECT Name,id,Industry,AnnualRevenue FROM Account ORDER BY AnnualRevenue ASC NULLS LAST
// null AnnualRevenue en sonda


SELECT Name,id,Industry,AnnualRevenue FROM Account ORDER BY AnnualRevenue ASC NULLS FIRST
// null AnnualRevenue en başta

SELECT Name,id,Industry,AnnualRevenue FROM Account ORDER BY AnnualRevenue DESC NULLS LAST
// AnnualRevenue azalarak ve null değerler sonda

SELECT Name,id,Industry,AnnualRevenue FROM Account WHERE AnnualRevenue!=null ORDER BY AnnualRevenue DESC
// AnnualRevenue null değer hiç yok. azalarak sıralama

LIMIT OFFSET

// LIMIT
SELECT Name,id,Industry,AnnualRevenue FROM Account ORDER BY AnnualRevenue DESC NULLS LAST LIMIT 5

// LIMIT OFFSET
SELECT Name,id,Industry,AnnualRevenue FROM Account ORDER BY AnnualRevenue DESC NULLS LAST LIMIT 10 OFFSET 10

// after 15th record it will get 16th and 17th. record
SELECT Name,id,Industry,AnnualRevenue FROM Account LIMIT 2 OFFSET 15

PARENT TO CHILD QUERY S and S__R

Only One Level parent to child is supported

Up to 20 related objects are supported

// Standard Object Plural name Contacts
// USE CHILD RELATIONSHIP NAME Check image below.

SELECT Name,id,Industry,AnnualRevenue,(SELECT firstName,lastName FROM Contacts) FROM Account WHERE AnnualRevenue >=50000000

SELECT Id, Name, Phone, Website, (SELECT Id, Subject FROM Cases),(SELECT Id, Name, Phone FROM Contacts) FROM Account

// for Custom object Plural and  __r 
SELECT Id, Name, Total_Income__c, Plan_Name__c, Total__c, (SELECT Value__c, Target__c, Name, Id FROM Expenses__r) FROM BudgetPlanMD__c

CHILD TO PARENT QUERY NO S

5 Level of parent allowed up to 55 related objects are supported.

// Standard Objects NO S Parent Object API.Field Name
SELECT firstName,LastName,Account.Name FROM Contact


// Custom Objects NO S  __r.Field Name
SELECT Value__c, Target__c, Name, Id,BudgetPlanMD__r.Name FROM Expense__c



SELECT firstName,lastName,Phone,Account.Name,Account.Phone,Account.AnnualRevenue FROM Contact

SELECT firstName,lastName,Phone,Account.Name,Account.Phone,Account.AnnualRevenue,Account.BillingState FROM Contact WHERE Account.Name LIKE '%United%'



SELECT Id, Account.id, Account.Name, Account.AnnualRevenue, (SELECT ContactId,Role, OpportunityId FROM OpportunityContactRoles) FROM Opportunity

    }
   
    System.debug('------------');
    
}

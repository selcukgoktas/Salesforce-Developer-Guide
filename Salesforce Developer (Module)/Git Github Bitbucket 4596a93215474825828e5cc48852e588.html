// after download you can use 

git --version

// help file

git --help

// to add empty git repo to your project folder. 
// .git folder

git init

// check the staging area 

git status

// setup global user
git config --global user.name "Selcuk Goktas"
git config --global user.email "info@selcukgoktas.com"

// check global user
git config --global user.name

// create short cuts with global alias
git config --global alias.p 'push'

// show logs one line
git log --pretty=oneline

// example git short cuts with oneline  l character is shortcut

git config --global alias.l 'log --pretty=oneline'
git l






// add files to staging area
git add .

// commit işlemi ile local repo

git commit -am "Commit Mesajınız"




// show the git changes
git log

// if you want to see the difference between the changes
git diff

git diff filename

// file delete process

// ---- 1st method delete from working directory by click del button. then
// don't use manuel delete or rename use git for this changes.

git status

// delete process will go to the stage area
git add filename
git rm filename

git commit -m "filename deleted"

// 2nd method delete via git

git rm filename
git status
git commit -m "filename deleted"


// if you want to delete folder   -r means recursive
git rm -r Folder/

git commit -m "folder deleted"


// manual move or rename files.

git status
git add .
git commit -m "filenameold --> filename new"


// git move or rename

git mv filename1 filename2

git status
git commit -m " file name is changed"



// before stage area you may turn to changes back. its like a ctrl + z

git status
git checkout -- filename


// After stage if already git add.  
git status
git reset HEAD filename
git checkout -- filename


// after commit change version to return back to specific version 
// DONT FORGET -- .
git checkout HASH_CODE_OF_THE_VERSION -- .

git status
git commit -m "version 2 copied now"

REMOTE GitHub


// connect github repo with our git system githubrepo is alias
git remote add githubrepo https://github.com/selcukgoktas/selcukgoktas.git

// to remove git remote
git remote remove url

// to list remote repos
git remote

// githubrepo  remote repo master local repo. 
// u all files. 
git push -u githubrepo master

// from local master to remote Dev branch. url is inside the githubrepo is url


// if there is no any remote repo like dev this will create 
// push the specific branch
git push githubrepo master:dev
git push origin localBranchName:remoteBranchName


// push from local master to remote dev. origin is remote url.
git push origin master:dev

// pull from the origin.
git pull origin master

If your Local branch and remote branch is the same name then you can just do it:

git push origin branchName

When your local and remote branch name is different then you can just do it:

git push origin localBranchName:remoteBranchName

Banches

// list branches ( locale )
git branch -v

// list branches ( remote)
git branch -r

// list branches ( all)
git branch -a
git branch --all

// Create local git branch
git branch newbranch

// delete local git branch
git branch -D newbranch

// you can create remote branch via github
1. click branch 
2. write name of the branch


// change branch
git checkout branchname

// check difference between two branch ( master brachname )
git diff master brachname

// merge from brachname with master ( you need to be in master )
git merge branchname

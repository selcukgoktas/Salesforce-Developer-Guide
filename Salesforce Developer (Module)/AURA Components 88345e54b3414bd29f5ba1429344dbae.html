Button Click add more Javascript ( Counter Example) 

<aura:component implements="force:appHostable,flexipage:availableForAllPageTypes,flexipage:availableForRecordHome,force:hasRecordId,forceCommunity:availableForAllPageTypes,force:lightningQuickAction" access="global">

   
    <aura:handler name="init" value="{!this}" action="{!c.doInit}"/>
    <aura:attribute name="newAcc" type="Account"
                     default="{'sobjectType':'Account',
                              'Name':'Test ACC AURA comp'
                              } "/>
    
    <aura:attribute name="arrayDemo" type="String[]"
                    default ="['tt1','werwer','sdffd','345345','35345']" />

     <aura:attribute name="helloMsg" type="String" />
    <aura:attribute name="buttonStatus" type="String"  default=" buton is not clicked"/>
     <aura:attribute name="clickcounter" type="String"  default="0" />
    
    
    <h1>{!v.newAcc.Name} </h1>
    
    <h2>{!v.arrayDemo[0]} </h2>
    
    <lightning:button label="ClickMe" onclick="{!c.handleClick}" />
    
    <h1>{!v.buttonStatus} </h1>
         
    <h1>Button {!v.clickcounter} times clicked </h1>
         
    
</aura:component>
({
	doInit : function(component, event, helper) {
		component.set("v.helloMsg"," Hello From Controller")
	}
    
    
    , handleClick: function(component, event, helper) {
        
        var counter = parseInt(component.get("v.clickcounter"));
        counter++;
        
		component.set("v.buttonStatus"," button is clicked");
        component.set("v.clickcounter",counter);
	}
})

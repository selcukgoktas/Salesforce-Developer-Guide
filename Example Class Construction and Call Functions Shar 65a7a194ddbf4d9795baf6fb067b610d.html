      return salesforce;  
    }
    
    public void setSalesforce(Integer num){
        all+=num;
       salesforce+=num;
    }
    public Integer getFullStack(){
       return fullstack;

    public void setFullStack(Integer num){
        all+=num;
      fullstack+=num;

    public static void reset(){
        all=0;

}

Delete

// Find Duplicate Values and Delete duplication. Just keep first one.

List<Account> multiAcc= [SELECT Name FROM Account WHERE Name LIKE '%Loop Account%'];
List<String> checkAcc= New List<String>();
List<Account> duplicateAcc=New List<Account>();

for (Account singleAcc:multiAcc){
     
    if(checkAcc.contains(singleAcc.Name)){
        // duplicate
         System.debug('Duplication : '+singleAcc);
         duplicateAcc.add(singleAcc);
    } 
    checkAcc.add(singleAcc.Name);
    System.debug('checkAcc : '+checkAcc);
   
}
System.debug('Duplicated Account count : ' + duplicateAcc.size());

delete duplicateAcc;

newAccs.add(acc);

Undelete

List<Account> multiAcc= [SELECT Name, isDeleted FROM Account WHERE Name LIKE '%Loop Account%' ALL ROWS];

List<Account> undelAccs= New List<Account>();


for (Account acc:multiAcc) {
    //System.debug('Account info :' +acc.Name + 'deleted Status : '+ acc.isDeleted);
    if (acc.isDeleted){
        undelAccs.add(acc);
    }
}

System.debug('Recyclebin Acc :'+undelAccs);

undelete undelAccs;
